```sql
SET @target_year := 2024;
SET @quarter := CONVERT('Q1,Q2' USING utf8mb4) COLLATE utf8mb4_0900_ai_ci;  -- 例如：'Q1,Q2'，也可传 'Q3' 或 'Q1,Q2,Q3,Q4'

WITH
-- 1) 产品大类全集：保证“同期没有某产品大类”也能出现在结果里
types AS (
  SELECT DISTINCT product_type
  FROM sales_details_t
  WHERE year IN (@target_year, @target_year - 1)
    AND FIND_IN_SET(quarter, @quarter) > 0
),

-- 2) 2024 汇总（业绩+回款）
curr AS (
  SELECT
    product_type,
    SUM(performance_amount) AS perf_curr,
    SUM(collection_amount)  AS coll_curr
  FROM sales_details_t
  WHERE year = @target_year
    AND FIND_IN_SET(quarter, @quarter) > 0
  GROUP BY product_type
),

-- 3) 2023 汇总（业绩+回款）
prev AS (
  SELECT
    product_type,
    SUM(performance_amount) AS perf_prev,
    SUM(collection_amount)  AS coll_prev
  FROM sales_details_t
  WHERE year = @target_year - 1
    AND FIND_IN_SET(quarter, @quarter) > 0
  GROUP BY product_type
),

-- 4) 2024 总业绩/总回款（用于占比分母 & “2024年全部业绩”）
total_curr AS (
  SELECT
    SUM(perf_curr) AS perf_sum_curr,
    SUM(coll_curr) AS coll_sum_curr
  FROM curr
)

SELECT
  t.product_type AS product_type,

  /* ===== 业绩 ===== */
  COALESCE(c.perf_curr, 0) AS perf_amount_curr_2024,
  ROUND(COALESCE(c.perf_curr, 0) / NULLIF(tc.perf_sum_curr, 0), 4) AS perf_share_curr_2024,
  ROUND(
    (COALESCE(c.perf_curr, 0) - COALESCE(p.perf_prev, 0))
    / NULLIF(COALESCE(p.perf_prev, 0), 0),
    4
  ) AS perf_yoy,

  /* ===== 回款 ===== */
  COALESCE(c.coll_curr, 0) AS coll_amount_curr_2024,
  ROUND(COALESCE(c.coll_curr, 0) / NULLIF(tc.coll_sum_curr, 0), 4) AS coll_share_curr_2024,
  ROUND(
    (COALESCE(c.coll_curr, 0) - COALESCE(p.coll_prev, 0))
    / NULLIF(COALESCE(p.coll_prev, 0), 0),
    4
  ) AS coll_yoy,

  /* ===== 2024 全部业绩（总额） ===== */
  COALESCE(tc.perf_sum_curr, 0) AS perf_total_2024_all

FROM types t
LEFT JOIN curr c ON c.product_type = t.product_type
LEFT JOIN prev p ON p.product_type = t.product_type
CROSS JOIN total_curr tc
ORDER BY perf_amount_curr_2024 DESC;


SHOW FULL COLUMNS FROM sales_details_t;

ALTER TABLE sales_details_t
MODIFY quarter VARCHAR(10)
CHARACTER SET utf8mb4
COLLATE utf8mb4_0900_ai_ci;
```
